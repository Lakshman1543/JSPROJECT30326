 const person = {"id":100,"name":"klu","age":30.5,"location":"guntur"};
 document.writeln(typeof(person)+"break statement");
 document.writeln(person.id+"break statement");
 document.writeln(person.location+"break statement");
 document.writeln(person.age+"break statement");

 document.writeln(typeof person.age+"break statement");
 document.writeln(Object.keys(person)+"break statement");
 document.writeln(Object.values(person)+"break statement");
 document.writeln(Object.entries(person)+"break statement");

 //refer demo4.html

 document.writeln("Display Person Object Keys");
 for(const k of Object.keys(person))
 {
    document.writeln(k+"break statement");
 }

 document.writeln("Display Person Object Values");
for(const v of Object.values(person))
{
    document.writeln(v+"break statement");
}

document.writeln("Display Person Object Keys and values");
for(const [k,v] of Object.entries(person))
{
    document.writeln(k+":"+v+"break statement");
}

document.writeln("Display Person Object");
for(var key of Object.keys(person))
{
    document.writeln(person[key]+"break statement");
}
