//map object - foreach method
            const mapobj1 = new Map();
            mapobj1.set("id",123);
            mapobj1.set("name","klef");
            mapobj1.set(1,100);

            document.writeln("Map object 1 size="+mapobj1.size+" ");

            document.writeln("Map object 1 based on values ");

            mapobj1.forEach( (v) => {
                document.writeln(v+" "); 
            })

            document.writeln("Map object 1 based on keys and values ");

            mapobj1.forEach((v,k) => {
                document.writeln(k+":"+v+" ");
            }
            )

            const mapobj2 = new Map(  [  ["id",101] , ["name","klef"]  ])   //constructor

            document.writeln("Map object 2 size="+mapobj2.size+" ");

            document.writeln("Map object 2 based on values ");

            mapobj2.forEach((v) => document.writeln(v+" ") );

            document.writeln("Map object 2 based on keys and values ");

            const view =  (v,k) => {
                document.writeln(k+":"+v+" ");
            }

            mapobj2.forEach(view); //view function is callback
