// map object - key value pair
        
        const demomap = new Map();

        demomap.set("id",101);
        demomap.set("name","mswd");
        demomap.set("age",34.5);
        demomap.set("status",true);
        demomap.set("name","klef"); // duplicate key - mswd will be changed to klef
        document.writeln("Map Size="+demomap.size+"  "); // 4
        document.writeln("Map Type="+typeof demomap.size+" "); //number

        document.writeln(demomap.get("id")+" "); // 101
        document.writeln(demomap.get("city")+" "); //undefined


        document.writeln(demomap.has("id")+" "); //true
        document.writeln(demomap.has("city")+" "); // false
        
        document.writeln("Deletion:"+demomap.delete("status")+"  "); // deletion of status
        document.writeln("Map Size="+demomap.size+" "); //3

        document.writeln("Display Map "); // Message
        for(const k of demomap.keys())
        {
            document.writeln(k);
        }

        document.writeln("break statement");

        document.writeln("Display Map Values "); // Message

        for(const v of demomap.values())
        {
            document.writeln(v);
        }

        document.writeln("break statement");

        document.writeln("Display Map Values based on keys");
        for(const key of demomap.keys())
        {
            document.writeln(demomap.get(key));
        }

        document.writeln("break statement");

        document.writeln("Display Map keys and values ");

        for(const[k,v] of demomap.entries())
        {
            document.writeln(k+":"+v);
        }

        document.writeln("break statement");
        demomap.clear();
        document.writeln("Map Size="+demomap.size+" ");
