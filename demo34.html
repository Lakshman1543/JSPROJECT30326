// Javascript class and object
    // class contains attributes/properties and method/member function
    class Student
    {
        setdata(id,name,status)
        {
            //sid,sname,sstatus are attributes in student class and attributes are also called properties
            this.sid = id;
            this.sname = name;
            this.sstatus = status;
        }
        getdata()
        {
            document.writeln("Student ID="+this.sid+"break statement");
            document.writeln("Student Name="+this.sname+"break statement");
            document.writeln("Student Status="+this.sstatus+"break statement");
        }
    }


    let s1 = new Student() //s1 is an object of student class
    s1.setdata(1,"KLU",true);
    s1.getdata()

    let s2 = new Student() // s2 is an object of student class 
    s2.setdata(2,"MSWD",false);
    s2.getdata()

    document.writeln(s1.sid+","+s1.sname+","+s1.sstatus);
    document.writeln("break statement");
    document.writeln(s2.sid+","+s2.sname+","+s2.sstatus);
