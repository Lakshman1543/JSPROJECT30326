// passing an array to the function
        function display(numbers)
        {
            let total = 0;
            for(element of numbers)
            {
                total = element + total;
            }
            return total;
        }

        const arr = [10,20,30,40,50]; //static array
        document.writeln("Array="+arr+" ");
        document.writeln("Output="+display(arr));
